// PWA INSTALAR
let deferredPrompt;
window.addEventListener('beforeinstallprompt', (e) => {
  e.preventDefault();
  deferredPrompt = e;
  document.getElementById("btnInstalar").style.display = "block";
});

document.getElementById("btnInstalar").addEventListener("click", () => {
  if(deferredPrompt){
    deferredPrompt.prompt();
  }
});

// SERVICE WORKER
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('sw.js');
}

// COMPARTIR
function compartir(){
  if(navigator.share){
    navigator.share({
      title:'FM MIX PARAGUAY',
      text:'Escuchá la radio en vivo',
      url:window.location.href
    });
  }else{
    alert("Copiá este link: "+window.location.href);
  }
}

// ===== TU CÓDIGO ORIGINAL =====

// ELEMENTOS
const radio=document.getElementById("radio");
const btn=document.getElementById("btn");
const horaEl=document.getElementById("hora");
const climaEl=document.getElementById("clima");
const monedasEl=document.getElementById("monedas");
const diariosEl=document.getElementById("diarios");

let activo=false;

// 🎧 RADIO
function toggleRadio(){
if(!activo){
radio.play().catch(()=>{});
btn.innerHTML="⏸";
activo=true;
}else{
radio.pause();
btn.innerHTML="▶";
activo=false;
}
}

// 🔁 RECONEXIÓN AUTOMÁTICA
setInterval(()=>{
if(activo && radio.paused){
radio.load();
radio.play().catch(()=>{});
}
},5000);

// 🕒 HORA
setInterval(()=>{
horaEl.innerHTML="🕒 "+new Date().toLocaleTimeString();
},1000);

// 🌤️ CLIMA
let ciudades=["Buenos Aires","Asuncion","Madrid"];
let i=0;

async function clima(){
try{
let res=await fetch("https://wttr.in/"+ciudades[i]+"?format=3");
let data=await res.text();
climaEl.innerHTML="🌍 "+data;
i=(i+1)%ciudades.length;
}catch{
climaEl.innerHTML="Error clima";
}
}
clima();
setInterval(clima,20000);

// 💵 MONEDAS
async function monedas(){
try{
let res=await fetch("https://open.er-api.com/v6/latest/USD");
let data=await res.json();

let ars=data.rates.ARS;
let pyg=data.rates.PYG;
let eur=data.rates.EUR;

monedasEl.innerHTML=`
💵 USD→ARS ${ars.toFixed(2)} 🇦🇷<br>
💵 USD→PYG ${pyg.toFixed(2)} 🇵🇾<br>
💱 ARS→PYG ${(pyg/ars).toFixed(2)}<br>
💶 EUR→ARS ${(ars/eur).toFixed(2)}
`;
}catch{
monedasEl.innerHTML="Error monedas";
}
}
monedas();
setInterval(monedas,3600000);
