let i = 0;

function mostrarNoticias(){

if(!window.noticiasLista || window.noticiasLista.length==0){
document.getElementById("noticiasBox").innerHTML = "📰 Cargando noticias...";
return;
}

let n = window.noticiasLista[i];

document.getElementById("noticiasBox").innerHTML = `
<img src="${n.img}">
<div class="noticias-texto">${n.titulo}</div>
`;

i = (i+1) % window.noticiasLista.length;
}

cargarNoticias();

setTimeout(()=>{
mostrarNoticias();
setInterval(mostrarNoticias,4000);
},3000);
